sbase
-----

	apt-get source coreutils

	git clone /usr/local/src/sbase
	cd sbase
	make
	./ls

	nm ls

	# LDFLAGS: -s
	vi config.mk
	make

	gdb ./ls
	run .
	b usage
	run -x
	list

	# CFLAGS: -g
	vi config.mk
	make

	gdb ./ls
	run .
	b usage
	run -x
	list
	s
	backtrace
	list
	print fmt
	n
	s
	finish
	n

	gdb -tui ./ls
